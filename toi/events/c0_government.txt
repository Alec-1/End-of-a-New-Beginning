#Questioning the status quo

country_event = {
  id = c0_government.2
  title = c0_government.2.t
  desc = {
    text = c0_government.2.d.m
    trigger = {
      has_government = despotic_autocracy
    }
  }
  desc = {
    text = c0_government.2.d.d
    trigger = {
      OR = {
        has_government = traditional_conservatism
        has_government = market_liberalism
        has_government = social_democracy
      }
    }
  }
  fire_only_once = yes

  is_triggered_only = yes
  option = {
    name = c0_government.2.a
  }
  option = {
    name = c0_government.2.b
  }
  option = {
    name = c0_government.2.c
  }
}

#1 : Opening a parliament

country_event = {
  id = c0_government.3
  title = c0_government.3.t
  desc = c0_government.3.d

  fire_only_once = yes

  is_triggered_only = yes
  immediate = {
    add_ideas = parliament
  }
  #
  option = {
    name = c0_government.3.a
  }
  option = {
    name = c0_government.3.b
    hidden = {
      set_country_flag = elected_parliament
      #Will displease nobles
    }
  }
  option = {
    name = c0_government.3.c
    hidden = {
      set_country_flag = unfair_parliament
      #Will deeply displease common people
    }
  }
}

  }

  }
