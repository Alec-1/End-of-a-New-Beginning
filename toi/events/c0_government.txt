      has_country_flag = questionthestatusquoc

      set_country_flag = reformdev1

  }

      remove_country_flag = questionthestatusquob

}

  desc = c0_government.4.desc

    has_country_flag = reformdev1
  }

  fire_only_once = yes

  mean_time_to_happen = {
    days = 100
    modifier = {
      factor = 0.1
      has_country_flag = quickreform
    }
    modifier = {
      factor = 2
      has_country_flag = slowreform
    }
  }

  immediate = {
    hidden_effect = {
      remove_country_flag = reformdev1
      set_country_flag = reformdev2
    }
  }

    name = c0_government.4.a
    add_ideas = reformstable1
  }

}
