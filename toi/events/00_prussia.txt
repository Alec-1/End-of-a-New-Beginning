#The German Question (Event Chain)
country_event = {
		id = prussia.5
		title = prussia.5.t
		desc = prussia.5.d
		
		fire_only_once = yes
		
		major = no
		
		trigger = {
				tag = PRS
				AND = {
					date > 1866.6.29
				}
		}
		
		#Großdeutschland Solution
		option = { 
				name = prussia.5.a
				add_ideas = PRS_GrossDE
				create_wargoal = { type = annex_everything target = HAN }
				create_wargoal = { type = annex_everything target = OLD }
				create_wargoal = { type = annex_everything target = MCK }
				create_wargoal = { type = annex_everything target = SAX }
				create_wargoal = { type = annex_everything target = THU }
				create_wargoal = { type = annex_everything target = HES }
				create_wargoal = { type = annex_everything target = BAV }
				create_wargoal = { type = annex_everything target = WUR }
				create_wargoal = { type = annex_everything target = BAD }
				add_state_claim = 152
				add_state_claim = 4
				add_state_claim = 102
				add_state_claim = 736
				add_state_claim = 153
				add_state_claim = 39
		}
		
		#Kleindeutschland Solution
		option = {
				name = prussia.5.b
				add_ideas = PRS_KleinDE
				hidden_effect = { PRS = { country_event = { days = 730 id = prussia.7 }}}
		}
}

					GER = { has_full_control_of_state = 102 }
					GER = { has_full_control_of_state = 736 }

		}
