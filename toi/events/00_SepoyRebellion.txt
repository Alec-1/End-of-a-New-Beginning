#Events about the Sepoy Rebellion
add_namespace = sepoyrebellion

#Sepoy breaking down
country_event = {
		id = sepoyrebellion.1
		title = sepoyrebellion.1.t
		desc = sepoyrebellion.1.d
		
		fire_only_once = yes
		
		trigger = {
				tag = SRS
				438 = { is_controlled_by = RAJ }
				437 = { is_controlled_by = RAJ }
				OR = {
					tag = SRS
					438 = { is_controlled_by = ENG }
					437 = { is_controlled_by = ENG }
				}
		}
		
		option = {
				id = sepoyrebellion.1.a
				add_timed_idea = { idea = SRS_Defense days = 180 }
		}
}

#Sepoy Army disbanded
country_event = {
		id = sepoyrebellion.6
		title = sepoyrebellion.6.t
		desc = sepoyrebellion.6.d
		
		fire_only_once = yes
		
		trigger = {
				date > 1857.5.11
				date < 1859.1.1
				tag = SRS
				num_divisions < 2
		}
		
		option = {
				id = sepoyrebellion.6.a
				add_timed_idea = { idea = SRS_ArmyDefeated days = 365 }
				remove_ideas = SRS_FreedomDemand
		}
}
